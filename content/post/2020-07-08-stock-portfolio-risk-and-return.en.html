library("tidyquant")
library("timetk")
library("ggplot2")
library("plyr")
#Data is downloaded using tq_get function from tidyquant
RSP <- tq_get("RSP", from = '2003-04-30', to = "2020-07-07", get = "stock.prices")

#Stock Prices are Plotted
RSP %>%
  ggplot(aes(x = date, y = adjusted)) +
  geom_line(size=0.5, color="steel blue") +
  ggtitle("Guggenheim Invest S&P 500 Pure Value ETF since 2003") +
  scale_x_date(date_breaks = "years", date_labels = "%Y") +
  labs(x = "Date", y = "Adjusted Price")
